SELECT I.SUMMARY AS TASKNAME, L.NAME AS LISTNAME
FROM ISSUES I
JOIN ISSUESLISTS L ON L.ID = I.ISSUESLIST_ID;


SELECT U.LASTNAME,  U.FIRSTNAME, I.SUMMARY AS TASKNAME 
FROM ISSUES I
JOIN USERS U ON U.ID = I.ISSUESLIST_ID;


SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) COUNT
FROM USERS U 
JOIN ISSUES I ON U.ID = I.ISSUESLIST_ID
GROUP BY 1,2;
